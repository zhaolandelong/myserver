    3. ▼提交问题(/cmapis/postQues)

      url: /cmapis/postQues
      type: post
      data: {
      name: 'cm',   //提问者姓名——string,required
      txt: '还是事实上',   //提问内容——string,required
      time: 1478184969145   //提问时间戳——number（可不传，在服务器取时间）
    }
      res: {
        code: 0,
        msg: 'success!'
    }
